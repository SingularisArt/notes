\newcommand{\enterEssayHeader}[1]{
    \nobreak\extramarks{}{Essay continued on next page\ldots}\nobreak{}
    \nobreak\extramarks{Essay (continued)}{Essay continued on next page\ldots}\nobreak{}
}

\newcommand{\exitEssayHeader}[1]{
    \nobreak\extramarks{Essay (continued)}{Essay continued on next page\ldots}\nobreak{}
    \stepcounter{#1}
    \nobreak\extramarks{Essay}{}\nobreak{}
}

\setcounter{secnumdepth}{0}
\newcounter{partCounter}
\newcounter{homeworkEssayCounter}
\setcounter{homeworkEssayCounter}{1}
\nobreak\extramarks{Essay}{}\nobreak{}

%
% Homework Essay Environment
%
% This environment takes an optional argument. When given, it will adjust the
% problem counter. This is useful for when the problems given for your
% assignment aren't sequential. See the last 3 problems of this template for an
% example.
%
\newenvironment{essay}[1][-1]{
    \ifnum#1>0
        \setcounter{homeworkEssayCounter}{#1}
    \fi
    \setcounter{partCounter}{1}
    \enterEssayHeader{homeworkEssayCounter}
}{
    \exitEssayHeader{homeworkEssayCounter}
}
