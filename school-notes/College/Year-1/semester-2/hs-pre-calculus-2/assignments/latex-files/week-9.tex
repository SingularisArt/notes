\begin{problem}
  Suppose that $\overrightarrow{\vec{v}} = \braket{2,-3}$.

  \begin{enumerate}
    \item Find $\Vert \overrightarrow{\vec{v}} \Vert$, the magnitude of
      $\overrightarrow{\vec{v}}$ \textbf{[$2.5$ points]}
    \item  Find the direction of the vector $\overrightarrow{\vec{v}}$.
      Represent the direction with an angle in degrees measured in standard
      position on the coordinate plane (so the positive $x$-axis is the initial
      side of the angle and the vector is the terminal side of the angle).
      \textbf{[$2.5$ points]}
  \end{enumerate}
\end{problem}

\begin{solution}
  \begin{enumerate}
    \item Find $\Vert \overrightarrow{\vec{v}} \Vert$, the magnitude of
      $\overrightarrow{\vec{v}}$ \textbf{[$2.5$ points]}

      \begin{align*}
        \Vert \overrightarrow{\vec{v}} \Vert &= \sqrt{2^{2} + (-3)^{2}} \\
                                             &= \sqrt{4 + 9} \\
        \Vert \overrightarrow{\vec{v}} \Vert &= \sqrt{13}
      .\end{align*}
    \item  Find the direction of the vector $\overrightarrow{\vec{v}}$.
      Represent the direction with an angle in degrees measured in standard
      position on the coordinate plane (so the positive $x$-axis is the initial
      side of the angle and the vector is the terminal side of the angle).
      \textbf{[$2.5$ points]}

      \begin{align*}
        \textrm{Direction of } \overrightarrow{\vec{v}} &= \tan^{-1} \left(-\frac{3}{2}\right) \\
        \textrm{Direction of } \overrightarrow{\vec{v}} &\approx -56.3^{\circ}
      .\end{align*}
  \end{enumerate}
\end{solution}

\begin{problem}
  Suppose that $\overrightarrow{\vec{v}} = \braket{2,-3}$ and
  $\overrightarrow{\vec{w}} = \braket{-4,5}$.

  \begin{enumerate}
    \item Find $2 \overrightarrow{\vec{v}} - \overrightarrow{\vec{w}}$
      \textbf{[$2.5$ points]}
    \item Find $\overrightarrow{\vec{v}} \cdot \overrightarrow{\vec{w}}$
      \textbf{[$2.5$ points]}
  \end{enumerate}
\end{problem}

\begin{solution}
  \begin{enumerate}
    \item Find $2 \overrightarrow{\vec{v}} - \overrightarrow{\vec{w}}$
      \textbf{[$2.5$ points]}

      \begin{align*}
        2 \overrightarrow{\vec{v}} - \overrightarrow{\vec{w}} &= \braket{4,-6} - \braket{-4,5} \\
        2 \overrightarrow{\vec{v}} - \overrightarrow{\vec{w}} &= \braket{0,-11}
      .\end{align*}

    \item Find $\overrightarrow{\vec{v}} \cdot \overrightarrow{\vec{w}}$
      \textbf{[$2.5$ points]}

      \begin{align*}
        \overrightarrow{\vec{v}} \cdot \overrightarrow{\vec{w}} &= (2 \times -4) + (-3 \times 5) \\
                                                                &= -8 + (-15) \\
        \overrightarrow{\vec{v}} \cdot \overrightarrow{\vec{w}} &= -23 \\
      .\end{align*}
  \end{enumerate}
\end{solution}

\begin{problem}
  Use the \textbf{dot product} to find the angle (in degrees) between the
  vectors $\overrightarrow{\vec{p}} = 3\overrightarrow{\vec{\imath}} +
  4\overrightarrow{\vec{\jmath}}$ and $\overrightarrow{\vec{q}} =
  \overrightarrow{\vec{\imath}} - 2\overrightarrow{\vec{\jmath}}$.
  (approximations are acceptable here but they must be denoted correctly.)
  \textbf{[$5$ points]}
\end{problem}

\begin{solution}
  \begin{align*}
    \Vert \overrightarrow{\vec{p}} \Vert &= \sqrt{3^{2} + 4^{2}} \\
                                         &= \sqrt{9 + 16} \\
                                         &= \sqrt{25} \\
    \Vert \overrightarrow{\vec{p}} \Vert &= 5 \\
    \Vert \overrightarrow{\vec{q}} \Vert &= \sqrt{1^{2} + (-2)^{2}} \\
                                         &= \sqrt{1 + 4} \\
    \Vert \overrightarrow{\vec{q}} \Vert &= \sqrt{5} \\
    \overrightarrow{\vec{p}} \cdot \overrightarrow{\vec{q}} &= (3 \times 1) + (4 \times -2) \\
                                                            &= 3 + -8 \\
    \overrightarrow{\vec{p}} \cdot \overrightarrow{\vec{q}} &= -5 \\
    \textrm{Angle between vectors } \overrightarrow{\vec{p}} \textrm{ and } \overrightarrow{\vec{q}} &= \cos (\theta) = \frac{-5}{5 \sqrt{5}} \\
                                                                                       \cos (\theta) &= -\frac{\sqrt{5}}{5} \\
                                                                                              \theta &= \cos^{-1} = \left(-\frac{\sqrt{5}}{5}\right) \\
    \textrm{Angle between vectors } \overrightarrow{\vec{p}} \textrm{ and } \overrightarrow{\vec{q}} &= 116.57^{\circ}
  .\end{align*}
\end{solution}
