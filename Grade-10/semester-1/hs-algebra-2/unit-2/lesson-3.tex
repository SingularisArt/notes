\subsubsection*{Factoring the Sums and Differences of Cubes}

\begin{example}[Factor $30c^3 - 20c^2 - 15c + 10$]
    \begin{align}
        2z^3 + 250 &= 2(z^3 + 125) \\
                   &= z^3 + 5^3 = (z + 5)(z^2 - z \times 5 + 5^2) \\
                   &= 2(z + 5)(z^2 - z \times 5 + 5^2) \\
    \end{align}
\end{example}
