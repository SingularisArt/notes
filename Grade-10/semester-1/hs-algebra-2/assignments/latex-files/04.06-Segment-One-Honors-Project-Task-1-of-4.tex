\subsubsection{$x^2 + 64$}

\paragraph{Factor}

\begin{align}
  a^2 - b^2 &= (\sqrt{a^2} + \sqrt{b^2})(\sqrt{a^2} - \sqrt{b^2}) \\
            &= (a + b)(a - b) \\
            &= x^2 + 64 = x^2 - (-64) \\
            &= (\sqrt{x^2} + \sqrt{-64})(\sqrt{x^2} + \sqrt{-64}) \\
            &= \boxed{(x + 8i)(x - 8i)}
.\end{align}

\paragraph{Check Work}

\begin{align}
  (x + 8i)(x - 8i) &= x^2 - 8xi + 8xi - 64i^2 \\
                   &= x^2 + 64
.\end{align}

\subsubsection{$16x^2 + 49$}

\paragraph{Factor}

\begin{align}
  a^2 - b^2 &= (\sqrt{a^2} + \sqrt{b^2})(\sqrt{a^2} - \sqrt{b^2}) \\
            &= (a + b)(a - b) \\
            &= 16x^2 + 49 = 16x^2 - (-49) \\
            &= (\sqrt{16x^2} + \sqrt{-49})(\sqrt{16x^2} + \sqrt{-49}) \\
            &= \boxed{(4x + 7i)(4x - 7i)}
.\end{align}

\paragraph{Check Work}

\begin{align}
  (4x + 7i)(4x - 7i) &= 16x^2 - 7xi + 7xi - 49i^2 \\
                     &= 16x^2 + 49
.\end{align}

\subsubsection{$(x + 9i)^2$}

\paragraph{Product}

\begin{align}
  (x + 9i)^2 &= (x + 9i)(x + 9i) \\
             &= x^2 + 9xi + 9xi + 81i^2 \\
             &= x^2 + 18xi + 81i^2
             &= x^2 + 18xi - 81
.\end{align}

\subsubsection{$(x - 2i)^2$}

\paragraph{Product}

\begin{align}
  (x - 2i)^2 &= (x - 2i)(x - 2i) \\
             &= x^2 - 2ix - 2ix + 4i^2 \\
             &= x^2 - 4xi - 4
.\end{align}

\subsubsection{$(x + (3 + 5i))^2$}

\paragraph{Product}

\begin{align}
  (x + (3 + 5i))^2 &= (x + (3 + 5i)(x + (3 + 5i))) \\
                   &= x^2 + 3x + 5xi + 3x + 9 + 15xi + 5xi + 15i + 25i^2 \\
                   &= x^2 + 10xi + 6x - 16 + 30i
.\end{align}
