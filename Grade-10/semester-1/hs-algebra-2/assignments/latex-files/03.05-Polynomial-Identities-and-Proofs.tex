\paragraph{Creating Polynomial Identity}

\begin{equation}
  \begin{split}
    (x - y)^2   &= x^2 - 2xy + y^2 \\
    (x^2 - 2xy + y^2) + (x^2 - 2xy + y^2) &= \boxed{2x^2 + 2y^2 - 4xy}
  \end{split}
.\end{equation}

\paragraph{Prove my Identity Works}

\begin{equation}
  \begin{split}
    x = 5, y &= 10 \\   
    (5 - 10)^2 &= 25 \\
    (5^2 - 2(5)(10) + 10^2) &= 25 \\
    25 + 25 &= \boxed{50} \\
    2(5)^2 + 2(10)^2 - 4(5)(10) &= \boxed{50}
  \end{split}
.\end{equation}
