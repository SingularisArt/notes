  \begin{align}
    (x + y)^{n} &= \sum_{k = 0}^{n} = {n \choose k}x^{n - k}y^{k} \Rightarrow {n \choose k} = \frac{n!}{k!(n - k)!} \\

                &= \boxed{x^{6} + 6x^{5} b + 20a^3b^2 + 20a^3b^3 + 15a^2b^{4} + 6ab^{5} + b^{6}}
  .\end{align}
\end{question}
