\begin{question}[Perform the following operations and prove closure.
  Show your work. \\
  $\frac{x}{x + 3} + \frac{x + 2}{x + 5}$]

  \begin{equation}
    \begin{split}
      \frac{x}{x + 3} + \frac{x + 2}{x + 5}
            &= \frac{x}{(x + 3)(x + 5)} + \frac{x + 2}{(x + 5)(x + 3)} \\
            &= \frac{x(x + 5)}{(x + 3)(x + 5)} + \frac{(x + 2)(x + 3)}{(x + 5)(x + 3)} \\
            &= \frac{x^2 + 5x}{(x + 3)(x + 5)} + \frac{x^2 + 5x + 6}{(x + 3)(x + 5)}\\
            &= \frac{2x^2 + 10x + 6}{(x + 3)(x + 5)} \\
            &= \boxed{\frac{2(x^2 + 5x + 3)}{(x + 3)(x + 5)}, \{ x \in \Re \| x \neq \{ -3, -5 \} \}}
    \end{split}
  .\end{equation}
\end{question}

\begin{question}[Perform the following operations and prove closure.
  Show your work. \\
  $\frac{x + 4}{x^2 + 5x + 6} \times \frac{x + 3}{x^2 - 16}$ ]

  \begin{equation}
    \begin{split}
      \frac{x + 4}{x^2 + 5x + 6} \times \frac{x + 3}{x^2 - 16}
            &= \frac{(x + 4)(x^2 - 16)}{(x^2 + 5x + 6)(x + 3)} \\
            &= \frac{(x + 4)(x^2 - 16)}{(x + 2)(x + 3)^2} \\
            &= \boxed{\frac{1}{(x + 2)(x - 4)}, \{ x \in \Re \| x \neq \{ -2, -3 \} \}}
    \end{split}
  .\end{equation}
\end{question}

\begin{question}[Perform the following operations and prove closure.
  Show your work. \\
  $\frac{2}{x^2 - 9} - \frac{3x}{x^2 - 5x + 6}$]

  \begin{equation}
    \begin{split}
      \frac{2}{x^2 - 9} - \frac{3x}{x^2 - 5x + 6}
            &= \frac{2(x^2 - 5x + 6)}{(x^2 - 9)(x^2 - 5x + 6)} - \frac{(x^2 - 9)}{(x^2 - 5x + 6)(2)} \\
            &= \boxed{- \frac{(x + 1)(3x + 4)}{(x + 3)(x - 2)(x - 3)}, \{ x \in \Re \| x \neq \{ 9, 3, 2 \} \}}
    \end{split}
  .\end{equation}
\end{question}

\begin{question}[Perform the following operations and prove closure.

  $\frac{x + 4}{x^2 - 5x + 6} \div \frac{x^2 - 16}{x + 3}$]

    \begin{split}
      \frac{x + 4}{x^2 - 5x + 6} \div \frac{x^2 - 16}{x + 3}
            &= \frac{x + 4}{x^2 - 5x + 6} \times \frac{x + 3}{x^2 - 16} \\
            &= \frac{(x + 4)(x + 3)}{(x^2 - 16)(x^2 - 5x + 5)} \\
            &= \boxed{\frac{x + 3}{(x - 2)(x - 3)(x - 4)}, \{ x \in \Re \| x \neq \{ 2, 3, -3,  \} \}}
    \end{split}
  .\end{equation}
\end{question}
