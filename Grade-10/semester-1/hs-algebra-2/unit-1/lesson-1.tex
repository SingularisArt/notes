\subsubsection*{Rational Exponents become $\rightarrow$ Radical Expressions}

The \textbf{numerator} of the \textbf{rational exponent} becomes the
\textbf{exponent} on the \textbf{radicand}. The \textbf{denominator} of the
\textbf{rational exponent} becomes the \textbf{index}, or \textbf{root}, of the
\textbf{radical}.

\begin{align}
    t^{\frac{3}{4}} = \sqrt[4]{t^3}
\end{align}

\subsubsection*{Radical Expressions become $\rightarrow$ Rational Exponents}

The \textbf{exponent} on the \textbf{radicand} becomes the \textbf{numerator} of the
\textbf{rational exponent}. The \textbf{index}, or \textbf{root}, of the \textbf{radical}
becomes the \textbf{denominator} of the \textbf{rational exponent}.

\begin{align}
    \sqrt[8]{w^5} = w^{\frac{5}{8}}
\end{align}
