\chapter{Radical and Polynomial Operations}

\lesson{1}{Sep 07 2021 Tue}{Rational Exponents}

In this lesson, I go over how you:

\begin{itemize}
	\item Convert radical expressions to rational exponents.
	\item Convert rational exponents to radical expressions.
\end{itemize}

\begin{definition}[Simplifying radical expressions]
	Rational expressions can be written as radical exponents.

	\[ t^{\frac{3}{4}} = t^{\frac{3}{4}} \times t^{\frac{3}{4}} \times t^{\frac{3}{4}} .\]
	\[ \sqrt[4]{t^3} = \sqrt[4]{t} \times \sqrt[4]{t} \times \sqrt[4]{t}  .\] 
	\[ t^{\frac{3}{4}} = \sqrt[4]{t^3} .\] 
\end{definition}

\begin{definition}[Simplifying radical expressions]
	Radical expressions can be written as rational exponents.

	\[ \sqrt[5]{x^3} = \sqrt[5]{x} \times \sqrt[5]{x} \times \sqrt[5]{x} .\]
	\[ x^{\frac{3}{5}} = x^{\frac{1}{5}} \times x^{\frac{1}{5}} \times x^{\frac{1}{5}} .\]
	\[ x^{\frac{3}{5}} = \sqrt[5]{x^3} .\] 
\end{definition}

\subsection{Quiz Questions}

\begin{question}
	Which of the following is the radical expression of $4d^{\frac{3}{8}}?$
\end{question}

\begin{answer}
	$\sqrt[8]{4d^3}$
\end{answer}

\begin{question}
	Which of the following is the radical expression of $4d^{\frac{3}{8}}?$
\end{question}

\begin{answer}
	$4\sqrt[8]{d^3}$
\end{answer}

\begin{question}
	Which of the following is the rational exponent expression of $\sqrt[4]{f}$
\end{question}

\begin{answer}
	$f^{\frac{1}{4}}$
\end{answer}

\begin{question}
	Which of the following is the rational exponent expression of $\sqrt[3]{4n}$
\end{question}

\begin{answer}
	$4n^{\frac{1}{3}}$
\end{answer}

\begin{question}
	Which of the following is the simplified form of $\sqrt[5]{x} \times \sqrt[5]{x} \times \sqrt[5]{x} \times \sqrt[5]{x}$
\end{question}

\begin{answer}
	$x^{\frac{4}{5}}$
\end{answer}
