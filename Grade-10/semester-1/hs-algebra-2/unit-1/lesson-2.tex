\begin{property}[Product Property]
    When multiplying $2$ of the same variable, the exponents are added. \bf{Fractions} must have common \bf{denominators} before being \bf{added} or \bf{subtracted}.
    
    \begin{align}
        a^{\frac{2}{5}} \times a^{\frac{1}{2}}  &= a^{\frac{4}{10}} \times a^{\frac{5}{10}} \\
                                                &= a^{\frac{9}{10}}
    \end{align}
\end{property}

\begin{property}[Quotient Property]
    When dividing $2$ of the same variable, the exponents are \bf{subtracted}. \bf{Fractions} must have been common \bf{denominators} before being \bf{added} or \bf{subtracted}.
    
    \begin{align}
        \frac{b^{\frac{5}{6}}}{b^{\frac{1}{4}}} &= \frac{b^{\frac{10}{12}}}{b^{\frac{3}{12}}} \\
                                                &= b^{\frac{10}{12} - \frac{3}{12}} \\
                                                &= b^{\frac{7}{12}}
    \end{align}
\end{property}

\begin{property}[Negative Exponents]
    When simplifying \bf{negative exponents}, take the \bf{reciprocal} of the \bf{expression} and make the \bf{exponent positive}.
    
    \begin{align}
        c^{-3}              &= \frac{1}{c^3} \\
        \frac{1}{d^{-4}}    &= \frac{1}{\frac{1}{d^4}} \\
                            &= 1 \div \frac{1}{d^4} \\
                            &= d^4
    \end{align}
\end{property}
