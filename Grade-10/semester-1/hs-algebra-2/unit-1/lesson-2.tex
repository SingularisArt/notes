\begin{property}[Product Property]
    When multiplying $2$ of the same variable, the exponents are added.
    \textbf{Fractions} must have common \textbf{denominators} before being
    \textbf{added} or \textbf{subtracted}.
    
    \begin{align}
        a^{\frac{2}{5}} \times a^{\frac{1}{2}}  &= a^{\frac{4}{10}} \times a^{\frac{5}{10}} \\
                                                &= a^{\frac{9}{10}}
    \end{align}
\end{property}

\begin{property}[Quotient Property]
    When dividing $2$ of the same variable, the exponents are \textbf{subtracted}.
    \textbf{Fractions} must have been common \textbf{denominators} before being
    \textbf{added} or \textbf{subtracted}.
    
    \begin{align}
        \frac{b^{\frac{5}{6}}}{b^{\frac{1}{4}}} &= \frac{b^{\frac{10}{12}}}{b^{\frac{3}{12}}} \\
                                                &= b^{\frac{10}{12} - \frac{3}{12}} \\
                                                &= b^{\frac{7}{12}}
    \end{align}
\end{property}

\begin{property}[Negative Exponents]
    When simplifying \textbf{negative exponents}, take the \textbf{reciprocal} of the
    \textbf{expression} and make the \textbf{exponent positive}.
    
    \begin{align}
        c^{-3}              &= \frac{1}{c^3} \\
        \frac{1}{d^{-4}}    &= \frac{1}{\frac{1}{d^4}} \\
                         &= 1 \div \frac{1}{d^4} \\
                         &= d^4
    \end{align}
\end{property}
