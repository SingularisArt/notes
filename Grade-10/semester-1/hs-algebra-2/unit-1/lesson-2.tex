Now you know that when like variables are multiplied, their exponents are added. But what happens when variables are divided:

\begin{property}[Quotient of Power Property]
	So, when you divide powers of the same base, subtract the exponents:

	\begin{align*}
		\frac{a^{m}}{a^{n}} = a^{m - n} \\
		\frac{a^{5}}{a^3} = \frac{\textrm{\sout{a}} \times \textrm{\sout{a}} \times \textrm{\sout{a}} \times a \times a}{\textrm{\sout{a}} \times \textrm{\sout{a}} \times \textrm{\sout{a}}} = a^2 = a^{5 - 3}
	.\end{align*}
\end{property}

\begin{example}[Example 1]
	Express the quotient of $\frac{r^{\frac{6}{7}}}{r^{\frac{2}{7}}}$ as a radical:

	\begin{align*}
		\frac{r^{\frac{6}{7}}}{r^{\frac{2}{7}}} = r^{\frac{6}{7} - \frac{2}{7}} = r^{\frac{4}{7}} = \sqrt[7]{r^{4}}
	.\end{align*}
\end{example}

\begin{property}[Quotient of Power Property]
	To raise a power to a power, multiply the exponents:

	\begin{align*}
		(a^{m})^{n} = a^{m \times n}
	.\end{align*}

	EXAMPLE:

	\[
		(a^2)^3 = (a \times a) \times (a \times a) \times (a \times a) = a^{6}
	.\] 
\end{property}

\begin{example}[Raising a Power to a Power]
	\[
		\left( c^{\frac{1}{2}} \right)^{\frac{1}{4}} \\
		c^{\frac{1}{2} \times \frac{1}{4}} \\
		c^{\frac{1}{8}} \\
		\sqrt[8]{c}
	.\]
\end{example}

\begin{property}[Negative Rational Exponents]
	You have learned all about positive integer exponents and rational, or
	fractional, exponents. Let's take a look at another type: \textbf{Negative
	Integer Exponents}.

	\begin{align*}
		4^{-1} = \frac{1}{4^1} = \frac{1}{4} \\
		4^{-2} = \frac{1}{4^2} = \frac{1}{16} \\
		4^{-3} = \frac{1}{4^3} = \frac{1}{64}
	.\end{align*}
\end{property}

\begin{example}[Negative Rational Exponents]
	\begin{align*}
		9^{-\frac{1}{2}} = \frac{1}{9^{\frac{1}{2}}} = \frac{1}{\sqrt{9}} = \frac{1}{3}
	.\end{align*}
\end{example}
