\subsection*{Different kinds of measurement tools}

\begin{definition}[Test tubes and test tube holder]
    Used to hold small amounts of a substance
\end{definition}

\begin{definition}[Dropper pipettes]
    Used to transport a measured volume of liquid
\end{definition}

\begin{definition}[Ring clamp and stand]
    A metal stand with a clamp used to support a container of material for
    testing
\end{definition}

\begin{definition}[Thermometer]
    An instrument used for measuring temperature
\end{definition}

\begin{definition}[Florence flask]
    A container used for uniform heating and boiling of liquids
\end{definition}

\begin{definition}[Beaker with glass stirring rod]
    A cylindrical glass container for holding substances and a glass rod for
    safe stirring of materials
\end{definition}

\begin{definition}[Erlenmeyer flask]
    A flat-bottomed flask for substances
\end{definition}

\begin{definition}[Balance scale]
    A device that compares the weight of an unknown object on one scale pan to
    the weight of a standard mass on the other scale pan
\end{definition}

\begin{definition}[Bunsen Burner]
    A heat source for heating substances in the laboratory
\end{definition}

\begin{definition}[Mortar and pestle]
    Used to crush ingredients or substances by grinding them into a powder
\end{definition}

\begin{definition}[Graduated cylinder]
    Equipment used to measure the volume of a liquid and some solids using
    water displacement
\end{definition}

\begin{definition}[Funnel]
    Used for guiding liquid or powder into a small opening
\end{definition}

\begin{definition}[Crucible]
    A ceramic or metal container used for melting or heating substances
\end{definition}
