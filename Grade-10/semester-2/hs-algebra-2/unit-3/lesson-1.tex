\begin{property}[Properties of Exponents]
  \label{prop:properties_of_exponents}

  The following are rules for exponents.

  For these, rules, we assume that the denominators involved aren't zero. \\

  \begin{center}
    \begin{tabular}{|l|c|r|}
      \hline
      Name & Rule & Example \\
      \hline
      Product & $x^{r} \times x^{s} = x^{r + s}$ & $x^{3} \times x^{5} = x^{8}$ \\
      \hline
      Quotient & $\frac{x^{r}}{x^{s}} = x^{r - s}$ & $\frac{a^{6}}{a^{2}} = a^{4}$ \\
      \hline
      Power of a Power & $\left(x^{r}\right)^{s} = x^{rs}$ & $\left(z^{2}\right)^{5} = z^{10}$  \\
      \hline
    \end{tabular}
  \end{center}
\end{property}

Simplify the following:

\begin{equation}
  \label{eqt:power_product_and_quotient_rules}
  \begin{split}
    \left(\frac{x^{3} - u^{-1}}{3z^{3}}\right)^{3} (xz^{5}) &= \frac{x^{-9} u^{3}}{3^{-3}z^{-9}} \times \frac{xz^{5}}{1} \\
        &= \frac{x^{-8} u^{3} z^{14}}{3^{-3}} \\
        &= \frac{3^{3} u^{3} z^{14}}{x^{8}} \\
        &= \frac{27u^{3} z^{14}}{x^{8}} \\
  \end{split}
.\end{equation}
