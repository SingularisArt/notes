\lesson{3}{Mar 18 2022 Fri (17:40:35)}{Use $i$ to rewrite imaginary numbers}{Unit 4}
\label{les__unit_:use_i_to_rewrite_imaginary_numbers}

Let's try to rewrite $\sqrt{-500}$.

\begin{equation}
  \label{spt:use_i_to_rewrite_imaginary_numbers}
  \begin{split}
    \sqrt{-1 \times 100 \times 5} &= \sqrt{-1} \times \sqrt{100} \times \sqrt{5} \\
                                  &= 10\sqrt{5}i
  \end{split}
.\end{equation}
