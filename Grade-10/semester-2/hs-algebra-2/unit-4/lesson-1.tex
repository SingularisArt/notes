\begin{theorem}[The Factor Theorem]
  \label{thrm:the_factor_theorem}

  If $f(x)$ is a polynomial of a degree $n \ge 1$ and $a \in \mathbb{R}$, then
  $(x - a)$ is a factor of $f(x)$, if $f(a) = 0$. Also, we can say, if $(x - a)$ 
  is a factor of polynomial $f(x)$, then $f(a) = 0$.
\end{theorem}

\begin{example}
  \label{exm:the_factor_theorem}

  Solve the equation $x^{3} - 4x^{2} + x + 6 = 0$ given that $3$ is a zero of
  $f(x) = x^{3} - 4x^{2} + x + 6$

  To test if $3$ is a zero/factor of $x^{3} - 4x^{2} + x + 6 = 0$, we'll use
  synthetic division:

  \begin{center}
    \polyhornerscheme[x=3]{x^{3}-4x^{2}+x+6}
  \end{center}

  As you can see, the remainder is $0$, which indicates that $x - 3$ is a factor
  of $x^{3} - 4x^{2} + x + 6 = 0$ and it tells us that $3$ is a factor of
  $x^{3} - 4x^{2} + x + 6 = 0$.
\end{example}
