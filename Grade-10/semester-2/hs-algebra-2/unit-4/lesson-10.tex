\lesson{10}{Mar 04 2022 Fri (23:55:49)}{Solve quadratic using square root}{Unit 4}

\begin{example}
  \label{exm:solve_quadratic_equation_using_square_root}

  Solve $(w - 9)^{2} - 36 = 0$, where $w$ is a real number.

  Simplify your answer as much as possible.

  We first rewrite the equation with only the squared expression on the left.

  \begin{equation}
    \label{eqt:step_1}
    \begin{split}
      (w - 9)^{2} = 36
    \end{split}
  .\end{equation}

  For the current problem, we have $w - 9 = \sqrt{36} \quad w - 9 = -\sqrt{36}$

  Solving and simplifying, we get the following:

  \begin{equation}
    \label{eqt:final_steps}
    \begin{split}
      w - 9 = \sqrt{36} \quad & \quad w - 9 = -\sqrt{36} \\
      w - 9 = 6 \quad & \quad w - 9 = -6 \\
      \boxed{w = 15} \quad & \quad \boxed{w = 3} \\
    \end{split}
  .\end{equation}
\end{example}
