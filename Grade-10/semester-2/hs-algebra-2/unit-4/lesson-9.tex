\begin{theorem}[Factor]
  \label{thrm:factor}

  A polynomial $P(x)$ has a factor $x - c \iff P(c) = 0$.
\end{theorem}

\begin{example}
  \label{exm:factor}

  We must determine whether $x - 2$ is a factor of $P(x) = 2x^{3} - 2x^{2} - 8$.

  So, to use the \textbf{Factor Theorem} (\ref{thrm:factor}), we must evaluate
  $P(2)$.

  \begin{equation}
    \label{eqt:factor}
    \begin{split}
      P(2)  &= 2(2)^{3} - 2(2)^{2} - 8 \\
            &= 16 - 8 - 8 \\
            &= 0
    \end{split}
  .\end{equation}

  So, $P(2) = 0$. By the \textbf{Theorem Factor} (\ref{thrm:factor}), we know
  that $x - 2$ is a factor of $P(x)$
\end{example}
