\lesson{4}{Mar 04 2022 Fri (19:51:45)}{Solve quadratic using square root}{Unit 4}

Consider this equation: $x^{2} = a$

\begin{itemize}
  \label{item:equation}

  \item When $a$ is positive, the equation has two solutions:
    $x = \sqrt{a} \quad x = -\sqrt{a}$ 

  \item When $a$ is negative, the equation has no solution. It only has
    imaginary solutions because the square root of a negative number and 
    is not a real number. It has the variable $i$, which is an imaginary number.
\end{itemize}

\begin{example}
  \label{exm:solv_quadratic_equation_using_square_root}

  Solve $x^{2} = 50$, where $x \in \mathbb{R}$.

  So, the equation $x^{2} = 50$ has two solutions:
  $x = \sqrt{50} \quad x = -\sqrt{50}$ 

  We can simplify:

  \begin{equation}
    \label{eqt:solv_quadratic_equation_using_square_root}
    \begin{split}
      \sqrt{50} = \sqrt{25 \times 2} = 5 \sqrt{2}
    \end{split}
  .\end{equation}

  So, our final answer is: $x = 5 \sqrt{2} \quad x = -5 \sqrt{2}$.
\end{example}
