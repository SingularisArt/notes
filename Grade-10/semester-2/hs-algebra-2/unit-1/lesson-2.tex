\lesson{2}{Feb 21 2022 Mon (12:24:44)}{Solving a linear equation by graphing}{Unit 1}

Let's look at this equation:

\begin{align}
  -4 = 5 - 3x
.\end{align}

Here's one method to solve it (by graphing):

\begin{marginfigure}
  \centering
  \incfig{solution-by-graphing}
  \sidecaption{$-4 = 5 - 3x$ Graphed}
  \label{fig:solution_by_graphing}
\end{marginfigure}

\begin{itemize}
  \label{item:solution_by_graphing}

  \item First, we write the equation with $0$ on one side

    \begin{equation}
      \begin{split}
        -4  &= 5 - 3x \\
        0   &= 9 - 3x
      \end{split}
    .\end{equation}

  \item Then, we graph the equation and find the \textbf{x-intercept}:

    \begin{table}[h]
      \centering
      \begin{tabular}{|c|c|c|}
        \hline
        $x$ & $y = -3x + 9$ & $(x, y)$ \\
        \hline
        $0$ & $y = -3 \times 0 + 9 = 9$ & $(0, 9)$ \\ 
        \hline
        $1$ & $y = -3 \times 1 + 9 = 6$ & $(1, 6)$ \\
        \hline
        $2$ & $y = -3 \times 2 + 9 = 3$ & $(2, 3)$ \\
        \hline
      \end{tabular}
      \caption{X-Y Table}
      \label{table:x_y_table}
    \end{table}

    We get that $x = 3$ is a solution to the original equation, which was
    $-4 = 5 -3x$
\end{itemize}
