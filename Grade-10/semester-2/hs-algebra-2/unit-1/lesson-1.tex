\lesson{1}{Mar 18 2022 Fri (16:25:45)}{Identify parallel and perpendicular lines}{Unit 1}
\label{les_1_unit_1:identify_parallel_and_perpendicular_lines}

\begin{definition}[Parallel]
  \label{def:parallel}

  When two lines don't intersect each other.
\end{definition}

\begin{definition}[Perpendicular]
  \label{def:perpendicular}

  When two lines cross at a right angle ($90^{\circ}$).
\end{definition}

\begin{definition}[Neither]
  \label{def:neither}

  When two lines cross, but not at right angles.
\end{definition}

Let's take a look at some important proprieties:

\begin{property}[Parallel Slope Property]
  \label{prop:parallel_slope_property}

  Two non-vertical lines are parallel if and only if they have the same slope.
\end{property}

\begin{property}[Perpendicular Slope Property]
  \label{prop:perpendicular_slope_property}

  Two non-vertical lines are perpendicular if and only if the product of their
  slopes is equal to $-1$.
\end{property}

\begin{property}[Neither Slope Property]
  \label{prop:perpendicular_slope_property}

  Two non-vertical lines are neither parallel and perpendicular if and only if
  there slope isn't the same and the products of their slopes aren't equal to
  $-1$.
\end{property}

\begin{example}
  \label{exm:identifying_if_an_equation_is_parallel_perpendicular_or_neither_1}

  Let's say we have the following system of linear equations:
  
  \begin{equation}
    \label{cse:system_of_linear_equations_1}
    \begin{cases}
      3x - 2y = 6 \\
      y = \frac{2}{3}x + 10
    \end{cases}
  .\end{equation}
  
  We need to find if they are \textbf{Parallel} (\ref{def:parallel})
  \textbf{Perpendicular} (\ref{def:perpendicular}) \textbf{Neither}
  (\ref{def:neither}).
  
  The key to do this re-writing these equations in the form $y = mx + b$ where $m$
  is the slope.
  
  Let's re-write the first equation in the form $y = mx + b$:
  
  \begin{equation}
    \label{edt:re_writing_the_first_equation_in_the_form_y_mx_b_1}
    \begin{alignedat}{3}
      3x - 2y &= 6 \qquad\qquad && \textrm{Given} \\
      -2y &= -3x + 6 \qquad\qquad && \textrm{Subtract $3x$ from both sides} \\
      y &= \frac{3}{2}x - 3 \qquad\qquad && \textrm{Divide both sides by $-2$} \\
    \end{alignedat}
  .\end{equation}

  So, our slope ($m$) is $\frac{3}{2}$. If we look at our second equation in the
  system of linear equations, we can see that it too has a slope of
  $\frac{3}{2}$, which makes this system of linear equations \textbf{Parallel}.
\end{example}

\begin{example}
  \label{exm:identifying_if_an_equation_is_parallel_perpendicular_or_neither_2}

  Let's say we have the following system of linear equations:
  
  \begin{equation}
    \label{cse:system_of_linear_equations_2}
    \begin{cases}
      2x - 5y = 20 \\
      y = -\frac{5}{2}x - 3
    \end{cases}
  .\end{equation}
  
  We need to find if they are \textbf{Parallel} (\ref{def:parallel}),
  \textbf{Perpendicular} (\ref{def:perpendicular}), or \textbf{Neither}
  (\ref{def:neither}).
  
  The key to do this re-writing these equations in the form $y = mx + b$ where $m$
  is the slope.
  
  Let's re-write the first equation in the form $y = mx + b$:
  
  \begin{equation}
    \label{edt:re_writing_the_first_equation_in_the_form_y_mx_b_2}
    \begin{alignedat}{3}
      2x - 5y &= 20 \qquad\qquad && \textrm{Given} \\
      -2x + 20 &= -5y \qquad\qquad && \textrm{Subtract $2x$ from both sides} \\
      y &= \frac{2}{5}x - 4 \qquad\qquad && \textrm{Divide both sides by $-5$} \\
    \end{alignedat}
  .\end{equation}

  So, our slope ($m$) is $\frac{2}{5}$. If we look at our second equation in the
  system of linear equations, we can see that it has a slope of $-\frac{5}{2}$,
  which is the reciprocal of $\frac{2}{5}$, which makes this system of linear
  equations \textbf{Perpendicular}
\end{example}

\begin{example}
  \label{exm:identifying_if_an_equation_is_parallel_perpendicular_or_neither_3}

  Let's say we have the following system of linear equations:

  \begin{equation}
    \label{cse:system_of_linear_equations_3}
    \begin{cases}
      3x + 8y = 12 \\
      2x - y = 7
    \end{cases}
  .\end{equation}

  We need to find if they are \textbf{Parallel} (\ref{def:parallel}),
  \textbf{Perpendicular} (\ref{def:perpendicular}), or \textbf{Neither}
  (\ref{def:neither}).
  
  The key to do this re-writing these equations in the form $y = mx + b$ where $m$
  is the slope.
  
  Let's re-write the first equation in the form $y = mx + b$:

  \begin{equation}
    \label{edt:re_writing_the_first_equation_in_the_form_y_mx_b_3}
    \begin{alignedat}{3}
      3x + 8y &= 12 \qquad\qquad && \textrm{Given} \\
      8y &= -3x + 12 \qquad\qquad && \textrm{Subtract $3x$ from both sides} \\
      y &= -\frac{3}{8}x + \frac{3}{2} \qquad\qquad && \textrm{Divide both sides by $8$} \\
    \end{alignedat}
  .\end{equation}

  So, our slope ($m$) is $-\frac{3}{8}$.

  Let's re-write the second equation in the form $y = mx + b$:

  \begin{equation}
    \label{edt:re_writing_the_second_equation_in_the_form_y_mx_b}
    \begin{alignedat}{3}
      2x - y &= 7 \qquad\qquad && \textrm{Given} \\
      -2x + 7 &= -y \qquad\qquad && \textrm{Subtract $2x$ from both sides} \\
      y &= 2x - 7 \qquad\qquad && \textrm{Divide both sides by $-1$} \\
    \end{alignedat}
  .\end{equation}

  So, our slope ($m$) is $\frac{2}{1}$.

  Since they aren't the same and they aren't opposite reciprocals, then that
  makes this system of linear equations \textbf{Neither}.
\end{example}
