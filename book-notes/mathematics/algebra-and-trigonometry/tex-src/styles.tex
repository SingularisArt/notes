%---------------%
% Chapter style %
%---------------%

\def\thickhrulefill{\leavevmode \leaders \hrule height 1ex \hfill \kern \z@}
\def\@makechapterhead#1{%
	%\vspace*{50\p@}%
	\vspace*{10\p@}%
	{\parindent \z@ \centering \reset@font
		\thickhrulefill\quad
		\scshape \@chapapp{} \thechapter
		\quad \thickhrulefill
		\par\nobreak
		\vspace*{10\p@}%
		\interlinepenalty\@M
		\hrule
		\vspace*{10\p@}%
		\Huge \bfseries #1\par\nobreak
		\par
		\vspace*{10\p@}%
		\hrule
		\vskip 20\p@
	}}
\def\@makeschapterhead#1{%
	\vspace*{10\p@}%
	{\parindent \z@ \centering \reset@font
		\thickhrulefill
		\par\nobreak
		\vspace*{10\p@}
		\interlinepenalty\@M
		\hrule
		\vspace*{10\p@}
		\Huge \bfseries #1\par\nobreak
		\par
		\vspace*{10\p@}
		\hrule
		\vskip 20\p@
	}}
\makeatother
